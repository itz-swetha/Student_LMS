1. CREATE DATABASE
use studentLMSDB

2. CREATE COLLECTIONS WITH VALIDATION
Students Collection

db.createCollection("students", {
  validator: {
    $jsonSchema: {
      bsonType: "object",
      required: ["name", "email", "course"],
      properties: {
        name: { bsonType: "string" },
        email: {
          bsonType: "string",
          pattern: "^.+@.+\\..+$"
        },
        course: { bsonType: "string" },
        age: {
          bsonType: "int",
          minimum: 16
        }
      }
    }
  }
})

Courses Collection

db.createCollection("courses", {
  validator: {
    $jsonSchema: {
      bsonType: "object",
      required: ["title", "duration", "instructor"],
      properties: {
        title: { bsonType: "string" },
        duration: {
          bsonType: "int",
          minimum: 1
        },
        instructor: { bsonType: "string" },
        level: {
          enum: ["Beginner", "Intermediate", "Advanced"]
        }
      }
    }
  }
})
Enrollments Collection

db.createCollection("enrollments", {
  validator: {
    $jsonSchema: {
      bsonType: "object",
      required: ["studentName", "courseTitle", "enrollDate"],
      properties: {
        studentName: { bsonType: "string" },
        courseTitle: { bsonType: "string" },
        enrollDate: { bsonType: "date" },
        status: {
          enum: ["ACTIVE", "COMPLETED", "DROPPED"]
        }
      }
    }
  }
})

3. INSERTION

5. UPDATE OPERATIONS

🔹 Update Student Age
db.students.updateOne(
  { name: "Swetha" },
  { $set: { age: 22 } }
)

🔹 Update Course Level
db.courses.updateOne(
  { title: "Java" },
  { $set: { level: "Advanced" } }
)

🔹 Update Enrollment Status
db.enrollments.updateOne(
  { studentName: "Arun" },
  { $set: { status: "ACTIVE" } }
)

6. VERIFY UPDATE (VERIFICATION STEP 2)

db.students.find({ name: "Swetha" }).pretty()
db.courses.find({ title: "Java" }).pretty()
db.enrollments.find({ studentName: "Arun" }).pretty()
